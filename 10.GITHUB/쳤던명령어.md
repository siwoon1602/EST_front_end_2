# 쳤던 명령어

```
mkdir test
cd test
git clone https://github.com/paullabkorea/test.git .
git add .
git commit -m "첫번째 커밋"
git push

# 여기서 부터 반복입니다.
git pull
# 파일 수정, 삭제, 추가
git add .
git commit -m "두번째 커밋"
git push

git pull
# 파일 수정, 삭제, 추가
git add .
git commit -m "두번째 커밋"
git push


# 간혹 log 보고 싶으실 때에는
git log
```

# 기본 리눅스 명령어

```
mkdir test
cd test
cd ..
rmdir test
touch test.txt
rm test.txt
ls
```
